var target = document.getElementById('minimal-left-2')
var input = document.getElementById('minimal-left-2-input')
var text = document.getElementById('minimal-left-2-text')
var scape = new ScapeRange({
  target: target,
  theme: 'minimal',
  min: 10,
  max: 100,
  targetInput: input, // 绑定的 input
  value: 20 // 初始值 20
})
scape.on('change', function (){
  text.innerHTML = this.val()
})
scape.trigger('change');
setTimeout(function() {
  scape.val(30).trigger('change') // val(value) 不会触发 change，如果需要触发，请 trigger('change')
}, 1000)
